# 安装步骤：
#
#   conda create -n sup python=3.10 -y
#   conda activate sup
#   cd backend
#
#   # 1) PyTorch：按显卡/CUDA 选择（示例为 CUDA 11.8）
#   pip install torch==2.2.2 torchvision==0.17.2 \
#       --index-url https://download.pytorch.org/whl/cu118
#
#   # 其他 CUDA 版本：
#   # CUDA 12.1: .../whl/cu121
#   # 仅 CPU:    pip install torch==2.2.2 torchvision==0.17.2
#
#   # 2) 本文件其余依赖
#   pip install -r requirements.txt
#
#   # 3) 修复 OpenCV 与 NumPy 兼容性（必做）
#   bash scripts/fix_sup_opencv.sh
#
#   # 4) 启动后端（必须在 backend 目录）
#   uvicorn app.main:app --reload --host 0.0.0.0 --port 8000
#
# 注意：
#   - 勿用 conda install opencv，会与 numpy 1.26 ABI 冲突
#   - LightGlue 通过 third_party/lightglue 加入 sys.path，无需 pip 安装
